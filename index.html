document.getElementById('break-time-display');

            // 타이머 상태 변수

            // 시계 눈금 그리기 함수
            function drawClockTicks() {
                const ticksGroup = document.getElementById('clock-ticks');
                for (let i = 0; i < 60; i++) {
                    const angle = i * 6; // 60분 = 360도이므로 1분당 6도
                    const rad = (angle - 90) * Math.PI / 180;
                    const isHour = i % 5 === 0; // 5분 단위 눈금은 굵고 길게
                    const innerRadius = isHour ? 85 : 90;
                    const outerRadius = 94;
                    const x1 = 100 + innerRadius * Math.cos(rad);
                    const y1 = 100 + innerRadius * Math.sin(rad);
                    const x2 = 100 + outerRadius * Math.cos(rad);
                    const y2 = 100 + outerRadius * Math.sin(rad);
                    const line = 
document.createElementNS('http://www.w3.org/2000/svg', 'line');
                    line.setAttribute('x1', x1);
                    line.setAttribute('y1', y1);
                    line.setAttribute('x2', x2);
                    line.setAttribute('y2', y2);
                    line.setAttribute('stroke', isHour ? '#9ca3af' : '#e5e7eb');
                    line.setAttribute('stroke-width', isHour ? '2' : '1');
                    ticksGroup.appendChild(line);
                }
            }
            // 시간 포맷 (MM:SS)
            function formatTime(seconds) {
                const mins = Math.floor(seconds / 60);
                const secs = seconds % 60;
                return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, 
'0')}`;
            }
            // 화면 업데이트 함수
            function updateDisplay() {
                // 텍스트 업데이트
                timeDisplay.textContent = formatTime(timeRemaining);
                
                // 모드에 따른 텍스트 변경
                modeDisplay.textContent = isWorkMode ? '집중 시간' : '휴식 시간';
                
                // 경과 시간에 따른 각도 계산
                const elapsedPercentage = (totalTime - timeRemaining) / totalTime;
                const currentAngle = elapsedPercentage * 360; // 0에서 시작해서 
360으로 증가
                
                // 진행 바늘(분침) 회전 - 시계방향으로 천천히 이동
                progressHand.setAttribute('transform', `rotate(${currentAngle} 100 
100)`);
                
                // 모드에 따른 색상 설정
                const currentColor = isWorkMode ? '#ef4444' : '#10b981';
                timeWedge.setAttribute('fill', currentColor);
                // 시계방향으로 하얀색 부분이 커지고 남은 색상 부채꼴을 렌더링하는 
Path 계산
                if (timeRemaining <= 0) {
                    timeWedge.setAttribute('d', ''); // 시간이 다 되면 숨김
                } else if (timeRemaining === totalTime || currentAngle <= 0.1) {
                    // 처음 시작할 때 꽉 찬 원으로 표시
                    timeWedge.setAttribute('d', 'M 100 100 L 100 10 A 90 90 0 1 1 
99.99 10 Z');
                } else {
                    // 현재 각도(바늘 위치)에서 12시 방향까지 시계방향으로 부채꼴 그림
                    const rad = (currentAngle - 90) * Math.PI / 180;
                    const startX = 100 + 90 * Math.cos(rad);
                    const startY = 100 + 90 * Math.sin(rad);
                    
                    // 남은 각도가 180도보다 크면 large-arc-flag를 1로 설정
                    const remainingAngle = 360 - currentAngle;
                    const largeArcFlag = remainingAngle > 180 ? 1 : 0;
                    
                    // Path 정의: 중앙(100,100) -> 바늘 끝 위치 -> 시계방향(1)으로 호
를 그림 -> 12시 위치(100,10) -> 중앙
                    const path = `M 100 100 L ${startX} ${startY} A 90 90 0 
${largeArcFlag} 1 100 10 Z`;
                    timeWedge.setAttribute('d', path);
                }
                // 문서 타이틀 업데이트
                document.title = `(${formatTime(timeRemaining)}) ${isWorkMode ? '집
중' : '휴식'} - 뽀모도로`;
            }
            // 타이머 시작
            function startTimer() {
                if (isRunning) return;
                
                isRunning = true;
                startBtn.classList.add('hidden');
                pauseBtn.classList.remove('hidden');
                
                settingsPanel.classList.remove('show');
                
                timerInterval = setInterval(() => {
                    if (timeRemaining > 0) {
                        timeRemaining--;
                        updateDisplay();
                    } else {
                        handleTimerComplete();
                    }
                }, 1000);
            }
            // 타이머 일시정지
            function pauseTimer() {
                if (!isRunning) return;
                
                isRunning = false;
                clearInterval(timerInterval);
                
                pauseBtn.classList.add('hidden');
                startBtn.classList.remove('hidden');
                document.title = `일시정지 - 뽀모도로`;
            }
            // 타이머 초기화 (현재 모드의 처음으로)
            function resetTimer() {
                pauseTimer();
                timeRemaining = isWorkMode ? workTime : breakTime;
                totalTime = timeRemaining;
                updateDisplay();
                document.title = '교육용 뽀모도로 타이머';
            }
            // 모드 전환 (집중 <-> 휴식)
            function switchMode() {
                isWorkMode = !isWorkMode;
                timeRemaining = isWorkMode ? workTime : breakTime;
                totalTime = timeRemaining;
                
                if (isWorkMode) {
                    app.classList.remove('mode-break');
                    app.classList.add('mode-work');
                } else {
                    app.classList.remove('mode-work');
                    app.classList.add('mode-break');
                }
                
                resetTimer();
            }
            // 타이머 완료 처리
            function handleTimerComplete() {

                document.body.appendChild(notif);
                
                setTimeout(() => {
                    notif.classList.remove('opacity-0');
                    notif.classList.add('opacity-100');

                setTimeout(() => {
                    notif.classList.remove('opacity-100');
                    notif.classList.add('opacity-0');
                    setTimeout(() => notif.remove(), 300);
                }, 4000);
            }
            // 세션 완료 점 업데이트
            function updateSessionDots() {

                    }
                    sessionDotsContainer.appendChild(dot);
                }
                
                const cycleText = document.querySelector('.text-xs.text-gray-400');
                if (completedSessions > 0 && completedSessions % 4 === 0) {
                    cycleText.textContent = `축하합니다! 4세션을 완료했습니다. 긴 휴식
(15~30분)을 권장합니다.`;
                    cycleText.classList.remove('text-gray-400');
                    cycleText.classList.add('text-indigo-500', 'font-medium');
                } else {
                    cycleText.textContent = '4번의 집중 후에는 긴 휴식을 취하세요';
                    cycleText.classList.add('text-gray-400');
                    cycleText.classList.remove('text-indigo-500', 'font-medium');
                }
            }
            // 알람 사운드 재생
            function playAlarm() {
                try {
                    const audioCtx = new (window.AudioContext || 
window.webkitAudioContext)();
                    const oscillator = audioCtx.createOscillator();
                    const gainNode = audioCtx.createGain();
                    
                    oscillator.connect(gainNode);
                    gainNode.connect(audioCtx.destination);
                    
                    oscillator.type = 'sine';
                    
                    oscillator.frequency.setValueAtTime(880, audioCtx.currentTime);
                    gainNode.gain.setValueAtTime(0, audioCtx.currentTime);
                    gainNode.gain.linearRampToValueAtTime(0.5, 
audioCtx.currentTime + 0.1);
                    
                    gainNode.gain.linearRampToValueAtTime(0.01, 
audioCtx.currentTime + 0.3);
                    
                    oscillator.frequency.setValueAtTime(880, audioCtx.currentTime + 
0.4);
                    gainNode.gain.linearRampToValueAtTime(0.5, 
audioCtx.currentTime + 0.4);
                    gainNode.gain.linearRampToValueAtTime(0, 
audioCtx.currentTime + 0.7);
                    
                    oscillator.start(audioCtx.currentTime);
                    oscillator.stop(audioCtx.currentTime + 0.8);
                } catch(e) {
                    console.log("Audio play error", e);
                }
            }

                const newWorkMin = parseInt(workTimeSlider.value);
                const newBreakMin = parseInt(breakTimeSlider.value);
                
                workTime = newWorkMin * 60;
                breakTime = newBreakMin * 60;
                
                settingsPanel.classList.remove('show');
                
                pauseTimer();
                timeRemaining = isWorkMode ? workTime : breakTime;
                totalTime = timeRemaining;
                updateDisplay();
                
                showNotification(`설정이 적용되었습니다.`);
            });
            
            // 페이지 이탈 경고
            window.addEventListener('beforeunload', (e) => {
                if (isRunning) {
                    e.preventDefault();
                    e.returnValue = '';
                }
            });
        });
